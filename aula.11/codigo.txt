 <h1>Sistema de multas</h1>
    Velocidae do carro: <input type="number" name="txtvel" id="txtvel"> km/l
    <input type="button" value="Verificar" onclick="calcular()">
    <div id="res"></div>
    <script> 
    
     function calcular() {
         var txtv = window.document.querySelector('input#txtvel')
         var res = window.document.querySelector('div#res')
         var vel = Number(txtv.value)
         res.innerHTML = `<p>Sua valocidade atual é de <strong>${vel}km/h.</strong></p>`
         if (vel > 60) {
             res.innerHTML += `<p>Você está <strong>MULTADO</strong> por excesso de velicodade!</p>`
         }
     }
        res.innerHTML += `<p>Dirija sempre com cinto de segurança.</p>`
    </script>
